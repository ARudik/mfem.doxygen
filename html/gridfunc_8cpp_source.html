00946 double GridFunction::ComputeH1Error(
00947    Coefficient *exsol, VectorCoefficient *exgrad,
00948    Coefficient *ell_coeff, double Nu, int norm_type) const
00949 {
00950    // assuming vdim is 1
00951    int i, fdof, dim, intorder, j, k;
00952    Mesh *mesh;
00953    const FiniteElement *fe;
00954    ElementTransformation *transf;
00955    FaceElementTransformations *face_elem_transf;
00956    Vector e_grad, a_grad, shape, el_dofs, err_val, ell_coeff_val;
00957    DenseMatrix dshape, dshapet, Jinv;
00958    Array<int> vdofs;
00959    IntegrationPoint eip;
00960    double error = 0.0;
00961 
00962    mesh = fes->GetMesh();
00963    dim = mesh->Dimension();
00964    e_grad.SetSize(dim);
00965    a_grad.SetSize(dim);
00966    Jinv.SetSize(dim);
00967 
00968    if (norm_type & 1)
00969       for (i = 0; i < mesh->GetNE(); i++)
00970       {
00971          fe = fes->GetFE(i);
00972          fdof = fe->GetDof();
00973          transf = mesh->GetElementTransformation(i);
00974          el_dofs.SetSize(fdof);
00975          dshape.SetSize(fdof, dim);
00976          dshapet.SetSize(fdof, dim);
00977          intorder = 2 * fe->GetOrder(); // <----------
00978          const IntegrationRule &ir = IntRules.Get(fe->GetGeomType(), intorder);
00979          fes->GetElementVDofs(i, vdofs);
00980          for (k = 0; k < fdof; k++)
00981             if (vdofs[k] >= 0)
00982                el_dofs(k) =   (*this)(vdofs[k]);
00983             else
00984                el_dofs(k) = - (*this)(-1-vdofs[k]);
00985          for (j = 0; j < ir.GetNPoints(); j++)
00986          {
00987             const IntegrationPoint &ip = ir.IntPoint(j);
00988             fe->CalcDShape(ip, dshape);
00989             transf->SetIntPoint(&ip);
00990             exgrad->Eval(e_grad, *transf, ip);
00991             CalcInverse(transf->Jacobian(), Jinv);
00992             Mult(dshape, Jinv, dshapet);
00993             dshapet.MultTranspose(el_dofs, a_grad);
00994             e_grad -= a_grad;
00995             error += (ip.weight * transf->Weight() *
00996                       ell_coeff->Eval(*transf, ip) *
00997                       (e_grad * e_grad));
00998          }
00999       }
01000 
01001    if (norm_type & 2)
01002       for (i = 0; i < mesh->GetNFaces(); i++)
01003       {
01004          face_elem_transf = mesh->GetFaceElementTransformations(i);
01005          int i1 = face_elem_transf->Elem1No;
01006          int i2 = face_elem_transf->Elem2No;
01007          intorder = fes->GetFE(i1)->GetOrder();
01008          if (i2 >= 0)
01009             if ( (k = fes->GetFE(i2)->GetOrder()) > intorder )
01010                intorder = k;
01011          intorder = 2 * intorder;  // <-------------
01012          const IntegrationRule &ir =
01013             IntRules.Get(face_elem_transf->FaceGeom, intorder);
01014          err_val.SetSize(ir.GetNPoints());
01015          ell_coeff_val.SetSize(ir.GetNPoints());
01016          // side 1
01017          transf = face_elem_transf->Elem1;
01018          fe = fes->GetFE(i1);
01019          fdof = fe->GetDof();
01020          fes->GetElementVDofs(i1, vdofs);
01021          shape.SetSize(fdof);
01022          el_dofs.SetSize(fdof);
01023          for (k = 0; k < fdof; k++)
01024             if (vdofs[k] >= 0)
01025                el_dofs(k) =   (*this)(vdofs[k]);
01026             else
01027                el_dofs(k) = - (*this)(-1-vdofs[k]);
01028          for (j = 0; j < ir.GetNPoints(); j++)
01029          {
01030             face_elem_transf->Loc1.Transform(ir.IntPoint(j), eip);
01031             fe->CalcShape(eip, shape);
01032             transf->SetIntPoint(&eip);
01033             ell_coeff_val(j) = ell_coeff->Eval(*transf, eip);
01034             err_val(j) = exsol->Eval(*transf, eip) - (shape * el_dofs);
01035          }
01036          if (i2 >= 0)
01037          {
01038             // side 2
01039             transf = face_elem_transf->Elem2;
01040             fe = fes->GetFE(i2);
01041             fdof = fe->GetDof();
01042             fes->GetElementVDofs(i2, vdofs);
01043             shape.SetSize(fdof);
01044             el_dofs.SetSize(fdof);
01045             for (k = 0; k < fdof; k++)
01046                if (vdofs[k] >= 0)
01047                   el_dofs(k) =   (*this)(vdofs[k]);
01048                else
01049                   el_dofs(k) = - (*this)(-1-vdofs[k]);
01050             for (j = 0; j < ir.GetNPoints(); j++)
01051             {
01052                face_elem_transf->Loc2.Transform(ir.IntPoint(j), eip);
01053                fe->CalcShape(eip, shape);
01054                transf->SetIntPoint(&eip);
01055                ell_coeff_val(j) += ell_coeff->Eval(*transf, eip);
01056                ell_coeff_val(j) *= 0.5;
01057                err_val(j) -= (exsol->Eval(*transf, eip) - (shape * el_dofs));
01058             }
01059          }
01060          transf = face_elem_transf->Face;
01061          for (j = 0; j < ir.GetNPoints(); j++)
01062          {
01063             const IntegrationPoint &ip = ir.IntPoint(j);
01064             transf->SetIntPoint(&ip);
01065             error += (ip.weight * Nu * ell_coeff_val(j) *
01066                       pow(transf->Weight(), 1.0-1.0/(dim-1)) *
01067                       err_val(j) * err_val(j));
01068          }
01069       }
01070 
01071    if (error < 0.0)
01072       return -sqrt(-error);
01073    return sqrt(error);
01074 }
