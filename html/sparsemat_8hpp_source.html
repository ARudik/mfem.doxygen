00241 inline double &SparseMatrix::SearchRow (const int row, const int col)
00242 {
00243    if (Rows)
00244    {
00245       RowNode *node_p;
00246 
00247       for (node_p = Rows[row]; 1; node_p = node_p->Prev)
00248          if (node_p == NULL)
00249          {
00250 #ifdef MFEM_USE_MEMALLOC
00251             node_p = NodesMem.Alloc();
00252 #else
00253             node_p = new RowNode;
00254 #endif
00255             node_p->Prev = Rows[row];
00256             node_p->Column = col;
00257             node_p->Value = 0.0;
00258             Rows[row] = node_p;
00259             break;
00260          }
00261          else if (node_p->Column == col)
00262          {
00263             break;
00264          }
00265       return node_p->Value;
00266    }
00267    else
00268    {
00269       int *Ip = I+row, *Jp = J;
00270       for (int k = Ip[0], end = Ip[1]; k < end; k++)
00271          if (Jp[k] == col)
00272             return A[k];
00273       mfem_error ("SparseMatrix::SearchRow (...)");
00274    }
00275    return A[0];
