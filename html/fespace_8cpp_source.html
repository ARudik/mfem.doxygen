00017 void FiniteElementSpace::DofsToVDofs (Array<int> &dofs) const
00018 {
00019    int i, j, size;
00020 
00021    if (vdim == 1)  return;
00022 
00023    size = dofs.Size();
00024    dofs.SetSize (size * vdim);
00025 
00026    switch(ordering)
00027    {
00028    case Ordering::byNODES:
00029       for (i = 1; i < vdim; i++)
00030          for (j = 0; j < size; j++)
00031             if (dofs[j] < 0)
00032                dofs[size * i + j] = -1 - ( ndofs * i + (-1-dofs[j]) );
00033             else
00034                dofs[size * i + j] = ndofs * i + dofs[j];
00035       break;
00036 
00037    case Ordering::byVDIM:
00038       for (i = vdim-1; i >= 0; i--)
00039          for (j = 0; j < size; j++)
00040             if (dofs[j] < 0)
00041                dofs[size * i + j] = -1 - ( (-1-dofs[j]) * vdim + i );
00042             else
00043                dofs[size * i + j] = dofs[j] * vdim + i;
00044       break;
00045    }
00046 }

00055       {
00056          int dof = dofs[i];

00059          else

00065       for (int i = 0; i < dofs.Size(); i++)
00066       {
00067          int dof = dofs[i];
00068          if (dof < 0)
00069             dofs[i] = -1 - ((-1-dof) * vdim + vd);
00070          else

00197    { ConstructRefinementData (k,num_c_dofs,type);  l = i; }

00200    Array<int> l_dofs, g_dofs;

00214 SparseMatrix * FiniteElementSpace::GlobalRestrictionMatrix
00215 (FiniteElementSpace * cfes, int one_vdim)
00216 {
00217    int k;
00218    SparseMatrix * R;
00219    DenseMatrix * r;
00220    Array<int> rows, cols, rs, cs;
00221 
00222    if (one_vdim == -1)
00223       one_vdim = (ordering == Ordering::byNODES) ? 1 : 0;
00224 
00225    mesh -> SetState (Mesh::TWO_LEVEL_COARSE);
00226    if (vdim == 1 || one_vdim == 1)
00227       R = new SparseMatrix (cfes -> GetNDofs(), ndofs);
00228    else
00229       R = new SparseMatrix (cfes -> GetVSize(), vdim*ndofs);
00230 
00231    for (k = 0; k < mesh -> GetNE(); k++)
00232    {
00233       cfes -> GetElementDofs (k, rows);
00234       mesh -> SetState (Mesh::TWO_LEVEL_FINE);
00235       r = LocalInterpolation (k, rows.Size(),
00236                               mesh -> GetRefinementType(k), cols);
00237       if (vdim == 1 || one_vdim == 1)
00238          R -> SetSubMatrixTranspose (rows, cols, *r, 1);
00239       else
00240       {
00241          int d, nr = rows.Size(), nc = cols.Size();
00242          cfes -> DofsToVDofs (rows);
00243          DofsToVDofs (cols);
00244          for (d = 0; d < vdim; d++)
00245          {
00246             rows.GetSubArray (d*nr, nr, rs);
00247             cols.GetSubArray (d*nc, nc, cs);
00248             R -> SetSubMatrixTranspose (rs, cs, *r, 1);
00249          }
00250       }
00251       mesh -> SetState (Mesh::TWO_LEVEL_COARSE);
00252    }
00253 
00254    return R;

00495 void FiniteElementSpace::GetElementDofs (int i, Array<int> &dofs) const
00496 {
00497    if (elem_dof)
00498    {
00499       elem_dof -> GetRow (i, dofs);
00500    }
00501    else
00502    {
00503       Array<int> V, E, Eo, F, Fo;
00504       int k, j, nv, ne, nf, nb, nfd, nd;
00505       int *ind, dim;
00506 
00507       dim = mesh->Dimension();
00508       nv = fec->DofForGeometry(Geometry::POINT);
00509       ne = (dim > 1) ? ( fec->DofForGeometry(Geometry::SEGMENT) ) : ( 0 );
00510       nb = fec->DofForGeometry(mesh->GetElementBaseGeometry(i));
00511       if (nv > 0)
00512          mesh->GetElementVertices(i, V);
00513       if (ne > 0)
00514          mesh->GetElementEdges(i, E, Eo);
00515       nfd = 0;
00516       if (dim == 3)
00517          if (fec->HasFaceDofs(mesh->GetElementBaseGeometry(i)))
00518          {
00519             mesh->GetElementFaces(i, F, Fo);
00520             for (k = 0; k < F.Size(); k++)
00521                nfd += fec->DofForGeometry(mesh->GetFaceBaseGeometry(F[k]));
00522          }
00523       nd = V.Size() * nv + E.Size() * ne + nfd + nb;
00524       dofs.SetSize(nd);
00525       if (nv > 0)
00526       {
00527          for (k = 0; k < V.Size(); k++)
00528             for (j = 0; j < nv; j++)
00529                dofs[k*nv+j] = V[k]*nv+j;
00530          nv *= V.Size();
00531       }
00532       if (ne > 0)
00533          // if (dim > 1)
00534          for (k = 0; k < E.Size(); k++)
00535          {
00536             ind = fec->DofOrderForOrientation(Geometry::SEGMENT, Eo[k]);
00537             for (j = 0; j < ne; j++)
00538                if (ind[j] < 0)
00539                   dofs[nv+k*ne+j] = -1 - ( nvdofs+E[k]*ne+(-1-ind[j]) );
00540                else
00541                   dofs[nv+k*ne+j] = nvdofs+E[k]*ne+ind[j];
00542          }
00543       ne = nv + ne * E.Size();
00544       if (nfd > 0)
00545          // if (dim == 3)
00546       {
00547          for (k = 0; k < F.Size(); k++)
00548          {
00549             ind = fec->DofOrderForOrientation(mesh->GetFaceBaseGeometry(F[k]),
00550                                               Fo[k]);
00551             nf = fec->DofForGeometry(mesh->GetFaceBaseGeometry(F[k]));
00552             for (j = 0; j < nf; j++)
00553             {
00554                if (ind[j] < 0)
00555                   dofs[ne+j] = -1 - ( nvdofs+nedofs+fdofs[F[k]]+(-1-ind[j]) );
00556                else
00557                   dofs[ne+j] = nvdofs+nedofs+fdofs[F[k]]+ind[j];
00558             }
00559             ne += nf;
00560          }
00561       }
00562       k = nvdofs + nedofs + nfdofs + bdofs[i];
00563       for (j = 0; j < nb; j++)
00564       {
00565          dofs[ne+j] = k + j;
00566       }
00567    }
00568 }
