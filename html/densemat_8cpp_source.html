00246 {

00257    {
00258       const double *d = data;
00259       double E = d[0] * d[0] + d[1] * d[1] + d[2] * d[2];
00260       double G = d[3] * d[3] + d[4] * d[4] + d[5] * d[5];
00261       double F = d[0] * d[3] + d[1] * d[4] + d[2] * d[5];
00262       return sqrt(E * G - F * F);
00263    }
00264    mfem_error ("DenseMatrix::Weight()");
