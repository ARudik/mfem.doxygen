00071    int nd = el.GetDof();
00072    int dim = el.GetDim();
00073    double w;
00074 
00075    elmat.SetSize(nd);
00076    dshape.SetSize(nd,dim);
00077    dshapedxt.SetSize(nd,dim);
00078    invdfdx.SetSize(dim);
00079 
00080    int order;
00081    if (el.Space() == FunctionSpace::Pk)
00082       order = 2*el.GetOrder() - 2;
00083    else
00084       //  order = 2*el.GetOrder();
00085       order = 2*el.GetOrder() + dim - 1;
00086 
00087    const IntegrationRule *ir;
00088    if (el.Space() == FunctionSpace::rQk)
00089       ir = &RefinedIntRules.Get(el.GetGeomType(), order);
00090    else

00092

00094    for (int i = 0; i < ir->GetNPoints(); i++)
00095    {
00096       const IntegrationPoint &ip = ir->IntPoint(i);
00097       el.CalcDShape(ip, dshape);
00098 
00099       Trans.SetIntPoint (&ip);
00100       CalcInverse(Trans.Jacobian(), invdfdx);
00101       w = Trans.Weight() * ip.weight;
00102       Mult(dshape, invdfdx, dshapedxt);
00103       if (Q)
00104       {
00105          w *= Q->Eval(Trans, ip);
00106          AddMult_a_AAt(w, dshapedxt, elmat);

00110          MQ->Eval(invdfdx, Trans, ip);

00164   Vector &flux)
00165 {
00166    int i, j, k, nd, dim, order;
00167    double energy, co;
00168 
00169    nd = fluxelem.GetDof();
00170    dim = fluxelem.GetDim();

00172    shape.SetSize(nd);
00173    pointflux.SetSize(dim);
00174    if (MQ)
00175    {
00176       invdfdx.SetSize(dim);
00177       vec.SetSize(dim);
00178    }
00179 
00180    order = 2 * fluxelem.GetOrder(); // <--
00181    const IntegrationRule *ir = &IntRules.Get(fluxelem.GetGeomType(), order);
00182 
00183    energy = 0.0;
00184    for (i = 0; i < ir->GetNPoints(); i++)
00185    {
00186       const IntegrationPoint &ip = ir->IntPoint(i);
00187       fluxelem.CalcShape(ip, shape);
00188 
00189       pointflux = 0.0;
00190       for (k = 0; k < dim; k++)
00191          for (j = 0; j < nd; j++)
00192             pointflux(k) += flux(k*nd+j)*shape(j);
00193 
00194       Trans.SetIntPoint (&ip);
00195       co = Trans.Weight() * ip.weight;
00196 
00197       if (Q)
00198          co *= Q->Eval(Trans, ip) * ( pointflux * pointflux );
00199       else
00200       {
00201          MQ->Eval(invdfdx, Trans, ip);
00202          invdfdx.Mult(pointflux, vec);
00203          co *= ( pointflux * vec );
00204       }
00205 
00206       energy += co;
00207    }
00208 
00209    return energy;

00327 void VectorMassIntegrator::AssembleElementMatrix
00328 ( const FiniteElement &el, ElementTransformation &Trans,
00329   DenseMatrix &elmat )
00330 {
00331    int nd   = el.GetDof();
00332    int dim  = el.GetDim();
00333    int vdim;
00334 
00335    double norm;
00336 
00337    // Get vdim from the ElementTransformation Trans ?
00338    vdim = (VQ) ? (VQ -> GetVDim()) : ((MQ) ? (MQ -> GetVDim()) : (dim));
00339 
00340    elmat.SetSize(nd*vdim);
00341    shape.SetSize(nd);
00342    partelmat.SetSize(nd);
00343    if (VQ)
00344       vec.SetSize(vdim);
00345    else if (MQ)
00346       mcoeff.SetSize(vdim);
00347 
00348    const IntegrationRule *ir = IntRule;
00349    if (ir == NULL)
00350    {
00351       int order = 2 * el.GetOrder() + Trans.OrderW() + Q_order;
00352 
00353       if (el.Space() == FunctionSpace::rQk)
00354          ir = &RefinedIntRules.Get(el.GetGeomType(), order);
00355       else
00356          ir = &IntRules.Get(el.GetGeomType(), order);
00357    }
00358 
00359    elmat = 0.0;
00360    for (int s = 0; s < ir->GetNPoints(); s++)
00361    {
00362       const IntegrationPoint &ip = ir->IntPoint(s);
00363       el.CalcShape(ip, shape);
00364 
00365       Trans.SetIntPoint (&ip);
00366       norm = ip.weight * Trans.Weight();
00367 
00368       MultVVt(shape, partelmat);
00369 
00370       if (Q)
00371       {
00372          norm *= Q -> Eval (Trans, ip);
00373          partelmat *= norm;
00374          for (int k = 0; k < vdim; k++)
00375             elmat.AddMatrix (partelmat, nd*k, nd*k);
00376       }
00377       else if (VQ)
00378       {
00379          VQ -> Eval (vec, Trans, ip);
00380          for (int k = 0; k < vdim; k++)
00381             elmat.AddMatrix (norm * vec(k), partelmat, nd*k, nd*k);
00382       }
00383       else // (MQ != NULL) -- matrix coefficient
00384       {
00385          MQ -> Eval (mcoeff, Trans, ip);
00386          for (int i = 0; i < vdim; i++)
00387             for (int j = 0; j < vdim; j++)
00388                elmat.AddMatrix (norm * mcoeff(i,j), partelmat, nd*i, nd*j);
00389       }
00390    }
00391 }
