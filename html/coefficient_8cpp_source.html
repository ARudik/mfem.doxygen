00140 double ComputeLpNorm(double p, Coefficient &coeff, Mesh &mesh,
00141                      const IntegrationRule *irs[])
00142 {
00143    double norm = 0.0;
00144    ElementTransformation *tr;
00145 
00146    for (int i = 0; i < mesh.GetNE(); i++)
00147    {
00148       tr = mesh.GetElementTransformation(i);
00149       const IntegrationRule &ir = *irs[mesh.GetElementType(i)];
00150       for (int j = 0; j < ir.GetNPoints(); j++)
00151       {
00152          const IntegrationPoint &ip = ir.IntPoint(j);
00153          tr->SetIntPoint(&ip);
00154          double val = fabs(coeff.Eval(*tr, ip));
00155          if (p < numeric_limits<double>::infinity())
00156          {
00157             norm += ip.weight * tr->Weight() * pow(val, p);
00158          }
00159          else
00160          {
00161             if (norm < val)
00162                norm = val;
00163          }
00164       }
00165    }
00166 
00167    if (p < numeric_limits<double>::infinity())
00168    {
00169       // negative quadrature weights may cause norm to be negative
00170       if (norm < 0.)
00171          norm = -pow(-norm, 1. / p);
00172       else
00173          norm = pow(norm, 1. / p);
00174    }
00175 
00176    return norm;
00177 }
